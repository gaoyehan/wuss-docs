# Icon 图标

### 使用指南

在 page.json 中引入组件

```json
"usingComponents": {
  "w-icon": "wuss-weapp/w-icon/index",
}
```

	 class="wuss-icon-items"

  hover-stop-propagation="false"
  catchtap="copyIconName"
  data-name="{{ item.name }}"
	 wx:for="{{ iconData }}"
	 wx:key="index"
	>
		<w-icon
		 type="{{ item.name }}"
   />

	</view>
</scroll-view>
